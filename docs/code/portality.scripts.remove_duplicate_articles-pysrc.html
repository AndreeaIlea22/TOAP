 1  """ 
 2  This script will run through the entire catalogue, detecting duplicates 
 3  of articles and removing them. It could take a long time and you should 
 4  take great care running it - at minimum double check there is a recent 
 5  backup available, there should always be one anyway. 
 6  """ 
 7   
 8  from portality import models 
 9  from portality.article import XWalk 
10  from portality.core import app 
11  from datetime import datetime 
12  import time 
13   
14  if __name__ == "__main__": 
15      if app.config.get("SCRIPTS_READ_ONLY_MODE", False): 
16          print "System is in READ-ONLY mode, script cannot run" 
17          exit() 
18   
19      import argparse 
20      parser = argparse.ArgumentParser() 
21   
22      parser.add_argument("-g", "--ghost", help="specify if you want the articles being deleted not to be snapshot", action="store_true") 
23      parser.add_argument("-y", "--yes", help="are you running this script with output redirection e.g. DOAJENV=production python portality/scripts/remove_duplicate_articles.py >> ~/dedupe_`date +\%%F_\%%H\%%M\%%S`.log 2>&1 ? You really should.", action="store_true") 
24   
25      args = parser.parse_args() 
26   
27      if not args.yes: 
28          raise Exception('Please run this script with output redirection, see --help if needed, and also confirm you are running it with output redirection by passing -y on run.') 
29   
30      snapshot = not args.ghost 
31      snapshot_report = "with snapshotting" if snapshot else "without snapshotting" 
32   
33      start = datetime.now() 
34      count = 0 
35      print 'Starting {0} {snapshot} in 5 seconds, Ctrl+C to exit.'.format(start.isoformat(), snapshot=snapshot_report) 
36      time.sleep(5) 
37   
38      originals = []  # make sure we only delete FORWARD. So, if articles 
39      # A, B and C are duplicates, we stumble upon A first, and delete B and C. 
40      # We must then ensure we do not delete A if we chance upon B or C 
41      # (because they are still in Python memory even though they are not in ES). 
42      article_iterator = models.Article.iterall(page_size=5000) 
43      for a in article_iterator: 
44          originals.append(a.id) 
45          duplicates = XWalk().get_duplicate(a, all_duplicates=True) 
46          if duplicates: 
47              print '{0}:'.format(a.id),  # original: ... 
48              for d in duplicates: 
49                  if not d.id in originals: 
50                      if snapshot: 
51                          d.snapshot() 
52                      d.delete() 
53                      count += 1 
54                      print '{0},'.format(d.id)  # ... duplicate_deleted1, duplicate_deleted2, etc. 
55                      time.sleep(1) 
56              print 
57   
58      end = datetime.now() 
59      length = end - start 
60      print "Ended {0}. {1} articles deleted. Took {2}.".format(end.isoformat(), count, str(length))
