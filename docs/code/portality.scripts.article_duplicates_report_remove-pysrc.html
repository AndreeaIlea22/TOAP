 18      dupcount = 0 
 19      deleted_ids = set()

 22          scroll_query = query_override

 30          article = Article(_source=a)

 33              continue 
 34   
 35          duplicates = XWalk.discover_duplicates(article, owner) 
 36          if duplicates: 
 37              doi_dups = [d.id for d in duplicates.get('doi', [])] 
 38              fulltext_dups = [d.id for d in duplicates.get('fulltext', [])] 
 39              doi_dups.sort(), fulltext_dups.sort()                            # Sort so we can visually compare the lists 
 40              print "\n{0}".format(article.id) 
 41              if doi_dups: 
 42                  print "\t{0} DOI duplicates: {1}".format(len(doi_dups), ", ".join(doi_dups)) 
 43              if fulltext_dups: 
 44                  print "\t{0} fulltext duplicates: {1}".format(len(fulltext_dups), ", ".join(fulltext_dups)) 
 45   
 46              # Only consider duplicates that appear in both lists (we should be cautious with deletes) 
 47              set_of_duplicates = set(doi_dups).intersection(set(fulltext_dups)) 
 48              dupcount += len(set_of_duplicates) + 1                     # The detected duplicates plus the article itself 
 49   
 50              if set_of_duplicates: 
 51                  print "\t{0} in intersection: {1}".format(len(set_of_duplicates), ", ".join(set_of_duplicates)) 
 52   
 53              if delete: 
 54                  for dup in set_of_duplicates: 
 55                      dup_art = article.pull(dup) 
 56                      if dup_art is not None: 
 57                          if snapshot: 
 58                              dup_art.snapshot() 
 59                          dup_art.delete() 
 60                          deleted_ids.add(dup) 
 61              else: 
 62                  deleted_ids = deleted_ids | set_of_duplicates 
 63   
 64      return dupcount, len(deleted_ids)
