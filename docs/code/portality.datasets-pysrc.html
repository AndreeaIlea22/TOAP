 23  CURRENCY_DEFAULT = '' 
 24  currency_options = [(CURRENCY_DEFAULT, CURRENCY_DEFAULT)] 
 25  currency_options_code_index = [] 
 26  currency_name_opts = [] 
 27   
 28  for code, country_info in countries:        # FIXME: a bit of a mess - would be better to have an object that just gave us the answers on demand 
 29      country_options.append((code, country_info['name'])) 
 30      country_options_two_char_code_index.append(code) 
 31      if 'currency_alphabetic_code' in country_info and 'currency_name' in country_info: 
 32          if country_info['currency_alphabetic_code'] not in currency_options_code_index: 
 33              # prevent duplicates in the currency options by checking if 
 34              # that currency has already been added - multiple countries 
 35              # use the same currency after all (GBP, EUR..) 
 36              currency_options.append( 
 37                  ( 
 38                      country_info['currency_alphabetic_code'], 
 39                      country_info['currency_alphabetic_code'] + ' - ' + country_info['currency_name'] 
 40                  ) 
 41              ) 
 42              currency_name_opts.append( 
 43                  ( 
 44                      country_info['currency_alphabetic_code'], 
 45                      country_info['currency_name'] 
 46                  ) 
 47              ) 
 48              currency_options_code_index.append(country_info['currency_alphabetic_code']) 
 49

 51  currency_name_map = dict(currency_name_opts)
